Lógica de Control JavaScript

// Algoritmo de Fibonacci usando ciclos controlados sin vectores
document.getElementById('btnCalcularFibonacci').addEventListener('click', () => {
    const meses = parseInt(document.getElementById('meses').value);
    const resultadoDiv = document.getElementById('resultadoFibonacci');
    
    if (isNaN(meses) || meses < 1 || meses > 40) {
        resultadoDiv.innerHTML = "Por favor, ingresa un número válido entre 1 y 40.";
        return;
    }

    let a = 0, b = 1, c;
    let HTML = "<strong>Población de Bacterias Calculada:</strong><br><br>";
    
    for (let i = 1; i <= meses; i++) {
        HTML += `Mes ${i}: <span style="color:var(--naranja-neon); font-weight:bold;">${b} bacterias</span><br>`;
        c = a + b;
        a = b;
        b = c;
    }
    
    resultadoDiv.innerHTML = HTML;
});

Nota del algoritmo: Se calcula mediante un ciclo controlado con variables simples sin usar vectores.
