// Fusión de ciclos controlados para evaluar volatilidad y zonas de soporte
function esZonaCritica(dia) {
    if (dia < 2) return false;
    for (let i = 2; i <= Math.sqrt(dia); i++) {
        if (dia % i === 0) return false;
    }
    return true; // Si el día es primo, se detecta volatilidad crítica
}

document.getElementById('btnProyectarMercado').addEventListener('click', () => {
    const crypto = document.getElementById('criptomoneda').value;
    const dias = parseInt(document.getElementById('diasProyeccion').value);
    const resultadoDiv = document.getElementById('resultadoCombinado');

    let a = 0, b = 1, c;
    let HTML = `<strong>Historial de Mercado para ${crypto}:</strong><br><br>`;

    for (let i = 1; i <= dias; i++) {
        let alerta = esZonaCritica(i) ? " [⚡ Resistencia Crítica]" : "";
        HTML += `Día ${i}: Valor Fibo: ${b} ${alerta}<br>`;
        c = a + b;
        a = b;
        b = c;
    }
    resultadoDiv.innerHTML = HTML;
});

Nota del algoritmo: Fusiona ciclos controlados: genera la serie matemática sin guardar vectores y detecta zonas de resistencia.
