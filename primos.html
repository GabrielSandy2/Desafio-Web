// Verificación de primalidad y desplazamiento de caracteres en tabla ASCII
function esPrimo(n) {
    if (n < 2) return false;
    for (let i = 2; i <= Math.sqrt(n); i++) {
        if (n % i === 0) return false;
    }
    return true;
}

// Cifrado sumando el valor de la clave al código ASCII
document.getElementById('btnEncriptar').addEventListener('click', () => {
    const texto = document.getElementById('textoProcesar').value;
    const clave = parseInt(document.getElementById('clavePrimo').value);
    const resultadoDiv = document.getElementById('resultadoPrimos');

    if (!esPrimo(clave)) {
        resultadoDiv.innerHTML = "Error: ¡La clave debe ser un número primo!";
        return;
    }

    let cifrado = texto.split('').map(char => 
        String.fromCharCode(char.charCodeAt(0) + clave)
    ).join('');

    resultadoDiv.innerHTML = `<strong>Texto Cifrado:</strong> ${cifrado}`;
});

Nota del algoritmo: Se verifica mediante divisiones consecutivas que la clave posea únicamente dos divisores exactos antes de alterar las cadenas.
